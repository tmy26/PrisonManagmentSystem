CREATE TABLE PRISONER(
id IDENTITY PRIMARY KEY,
fname VARCHAR(20) NOT NULL,
lname VARCHAR(20) NOT NULL,
sex VARCHAR(20) NOT NULL,
age int NOT NULL,
reason VARCHAR(50) NOT NULL
)


CREAT TABLE INFO(
id IDENTITY PRIMARY KEY,
reportPolice VARCHAR(1000) NOT NULL,
reportWitness1 VARCHAR(1000) NOT NULL,
reportWitness2 VARCHAR(1000),
reportWitness3 VARCHAR(1000),
dateIN
dateOUT
)


CREATE TABLE ADDITIONALINFO (
id IDENTITY PRIMARY KEY,
id int REFERENCES INFO(id),
id int REFERENCES PRISONER(id),
osudenVliza date,
osudenIzliza date
)
